## 🔄 配置访问方法

### 旧方法（已废弃）

```python
# ❌ 不再可用
import yaml
with open('backend/config/device_params.yaml', 'r') as f:
    device_params = yaml.safe_load(f)

# ❌ 不再可用
import json
with open('backend/config/intelligent_extraction_config.json', 'r') as f:
    extraction_config = json.load(f)
```

### 新方法（推荐）

```python
# ✅ 从数据库读取
from modules.database import DatabaseManager
from modules.database_loader import DatabaseLoader

db_manager = DatabaseManager("sqlite:///data/devices.db")
db_loader = DatabaseLoader(db_manager)

# 访问 device_params（原 device_params.yaml）
device_params = db_loader.get_config_by_key('device_params')

# 访问 intelligent_extraction（原 intelligent_extraction_config.json）
extraction_config = db_loader.get_config_by_key('intelligent_extraction')

# 获取所有配置
all_config = db_loader.load_config()
```

## 🎯 使用示例

### 示例1：访问设备参数配置

```python
import sys
sys.path.insert(0, 'backend')

from modules.database import DatabaseManager
from modules.database_loader import DatabaseLoader

# 初始化
db_manager = DatabaseManager("sqlite:///data/devices.db")
db_loader = DatabaseLoader(db_manager)

# 获取设备参数配置
device_params = db_loader.get_config_by_key('device_params')

# 访问具体设备类型
if device_params and 'device_types' in device_params:
    # 获取温度传感器参数
    sensor_params = device_params['device_types'].get('温度传感器', {})
    params_list = sensor_params.get('params', [])
    
    print(f"温度传感器参数配置：")
    for param in params_list:
        print(f"  - {param['name']}: 必填={param['required']}")
```

### 示例2：访问智能提取配置

```python
# 获取智能提取配置
extraction_config = db_loader.get_config_by_key('intelligent_extraction')

if extraction_config:
    # 访问设备类型列表
    device_types = extraction_config.get('device_type', {}).get('device_types', [])
    print(f"支持的设备类型: {device_types}")
    
    # 访问匹配权重
    matching_weights = extraction_config.get('matching', {}).get('weights', {})
    print(f"匹配权重配置: {matching_weights}")
    
    # 访问参数提取配置
    parameter_config = extraction_config.get('parameter', {})
    range_config = parameter_config.get('range', {})
    print(f"量程提取配置: {range_config}")
```

### 示例3：更新配置

```python
# 更新设备参数配置
device_params = db_loader.get_config_by_key('device_params')

# 修改配置
if device_params:
    # 添加新的设备类型
    if 'device_types' not in device_params:
        device_params['device_types'] = {}
    
    device_params['device_types']['新设备类型'] = {
        'keywords': ['关键词1', '关键词2'],
        'params': [
            {
                'name': '参数1',
                'type': 'string',
                'required': True,
                'options': ['选项1', '选项2']
            }
        ]
    }
    
    # 保存回数据库
    db_loader.update_config('device_params', device_params)
    print("✓ 配置更新成功")
```
