        def process_device(device):
            return extractor.extract_and_match(device)
        
        # 使用10个并发线程
        with concurrent.futures.ThreadPoolExecutor(max_workers=10) as executor:
            start_time = time.time()
            results = list(executor.map(process_device, devices))
            end_time = time.time()
        
        total_time = end_time - start_time
        avg_time_per_device = (total_time / len(devices)) * 1000
        
        assert avg_time_per_device < 1000, f"并发情况下平均处理时间 {avg_time_per_device:.2f}ms 超过 1000ms"
        assert all(r is not None for r in results), "部分请求失败"
        print(f"并发测试: {len(devices)}个设备, 总时间 {total_time:.2f}秒, 平均 {avg_time_per_device:.2f}ms/设备")

        assert accuracy >= 0.90, f"Top-3准确率 {accuracy:.2%} 低于 90%"
